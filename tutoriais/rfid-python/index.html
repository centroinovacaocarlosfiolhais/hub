#!/usr/bin/env python3
"""
Sistema Check-in/Check-out - CICF
Autor: Edgar (adaptado para tutorial)
"""

import time
from datetime import datetime
import sqlite3
import os
import RPi.GPIO as GPIO
from mfrc522 import SimpleMFRC522

# Configura√ß√µes
SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
DB_PATH = os.path.join(SCRIPT_DIR, "attendance.db")
print(f"Base de dados: {DB_PATH}")

# Configurar GPIO
GPIO.setwarnings(False)
GPIO.setmode(GPIO.BOARD)

# Inicializar leitor RFID
print("Inicializando leitor MFRC522...")
reader = SimpleMFRC522()
print("Leitor inicializado!")

print("=== SISTEMA CHECK-IN/CHECK-OUT CICF ===")
print("Aproxime o cart√£o NFC... (Ctrl+C para sair)\n")

# Criar tabela se n√£o existir
conn = sqlite3.connect(DB_PATH)
cursor = conn.cursor()

cursor.execute('''
    CREATE TABLE IF NOT EXISTS movimentos (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        uid TEXT NOT NULL,
        nome TEXT NOT NULL,
        tipo TEXT NOT NULL,
        data_hora TEXT NOT NULL,
        timestamp INTEGER NOT NULL
    )
''')

conn.commit()
conn.close()

# Controlo de presen√ßa em mem√≥ria
presentes = {}  # uid: hora_entrada

try:
    while True:
        print(".", end="", flush=True)
        id_cartao, texto = reader.read_no_block()
        
        if id_cartao is not None:
            uid_str = str(id_cartao)
            
            # Nome baseado no UID (ou usar dicion√°rio de utilizadores)
            nome = f"Utilizador_{uid_str[-6:]}"
            
            print(f"\n\nüîë TAG DETECTADA! UID: {uid_str}")
            if texto.strip():
                print(f"üìù Texto: '{texto.strip()}'")
            
            agora = datetime.now()
            data_hora_str = agora.strftime("%Y-%m-%d %H:%M:%S")
            timestamp = int(time.time())
            
            # Conectar √† base de dados
            conn = sqlite3.connect(DB_PATH)
            cursor = conn.cursor()
            
            if uid_str in presentes:
                # CHECK-OUT
                entrada = presentes[uid_str]
                tempo = agora - entrada
                print(f"üö™ CHECK-OUT ‚Üí {nome}")
                print(f"‚è∞ Entrou: {entrada.strftime('%H:%M:%S')}")
                print(f"‚è∞ Saiu:   {agora.strftime('%H:%M:%S')}")
                print(f"‚è±Ô∏è  Tempo:  {str(tempo).split('.')[0]}")
                
                # Registar sa√≠da
                cursor.execute('''
                    INSERT INTO movimentos (uid, nome, tipo, data_hora, timestamp)
                    VALUES (?, ?, ?, ?, ?)
                ''', (uid_str, nome, 'saida', data_hora_str, timestamp))
                
                # Remover do controlo
                del presentes[uid_str]
                
            else:
                # CHECK-IN
                print(f"üö™ CHECK-IN ‚Üí {nome}")
                print(f"‚è∞ Entrada: {agora.strftime('%H:%M:%S')}")
                
                # Registar entrada
                cursor.execute('''
                    INSERT INTO movimentos (uid, nome, tipo, data_hora, timestamp)
                    VALUES (?, ?, ?, ?, ?)
                ''', (uid_str, nome, 'entrada', data_hora_str, timestamp))
                
                # Adicionar ao controlo
                presentes[uid_str] = agora
            
            conn.commit()
            conn.close()
            
            print(f"üíæ Registado na base de dados!")
            print("-" * 40)
            
            # Pausa para evitar leituras m√∫ltiplas
            time.sleep(2)

except KeyboardInterrupt:
    print("\n\nüëã Sistema encerrado!")
    print(f"üìä Base de dados guardada em: {DB_PATH}")
    GPIO.cleanup()
